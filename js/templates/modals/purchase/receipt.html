<%= ob.polyT('purchase.receipt.additionalItems') %> <%= ob.currency.formatCurrency(additionalShippingPrice, viewingCurrency) %>
<% } else if (ob.listing.shippingOptions && ob.listing.shippingOptions.length) { %>
<%= ob.polyT('purchase.receipt.shipping') %> <%= ob.currency.formatCurrency(shippingPrice, viewingCurrency) %>
<% } %>

<%= ob.polyT('purchase.receipt.subtotal', { quantity }) %> <%= ob.currency.formatCurrency(subTotal, viewingCurrency) %>
